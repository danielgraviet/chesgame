3    /**
4     * A chessboard that can hold and rearrange chess pieces.
5     * <p>
6     * Note: You can add to this class, but you may not alter
7     * signature of the existing methods.
8     */
9    import java.util.Arrays;
10   import java.util.Objects;
11   
12   public class ChessBoard {
13   
14       private ChessPiece[][] board;
15       public ChessBoard() {
16           board = new ChessPiece[8][8];
17       }
18   
19       /**
20        * Adds a chess piece to the chessboard
21        *
22        * @param position where to add the piece to
23        * @param piece    the piece to add
24        */
25       public void addPiece(ChessPosition position, ChessPiece piece) {
26           board[position.getRow()-1][position.getColumn()-1] = piece;
27       }
28   
29       /**
30        * Gets a chess piece on the chessboard
31        *
32        * @param position The position to get the piece from
33        * @return Either the piece at the position, or null if no piece is at that
34        * position
35        */
36       public ChessPiece getPiece(ChessPosition position) {
37           return board[position.getRow()-1][position.getColumn()-1];
38       }
39   
40       /**
41        * Sets the board to the default starting board
42        * (How the game of chess normally starts)
43        */
44       public void resetBoard() {
45           // clear existing board
46           board = new ChessPiece[8][8];
47   
48           // set black and white pawns
49           for (int column = 1; column <= 8; column++) {
50               addPiece(new ChessPosition(2, column), new ChessPiece(ChessGame.TeamColor.WHITE, ChessPiece.PieceType.PAWN));
51               addPiece(new ChessPosition(7, column), new ChessPiece(ChessGame.TeamColor.BLACK, ChessPiece.PieceType.PAWN));
52           }
53   
54           // set rooks
55           addPiece(new ChessPosition(1, 1), new ChessPiece(ChessGame.TeamColor.WHITE, ChessPiece.PieceType.ROOK));
56           addPiece(new ChessPosition(1, 8), new ChessPiece(ChessGame.TeamColor.WHITE, ChessPiece.PieceType.ROOK));
57           addPiece(new ChessPosition(8, 8), new ChessPiece(ChessGame.TeamColor.BLACK, ChessPiece.PieceType.ROOK));
58           addPiece(new ChessPosition(8, 1), new ChessPiece(ChessGame.TeamColor.BLACK, ChessPiece.PieceType.ROOK));
59   
60           // set knights
61           addPiece(new ChessPosition(1, 2), new ChessPiece(ChessGame.TeamColor.WHITE, ChessPiece.PieceType.KNIGHT));
62           addPiece(new ChessPosition(1, 7), new ChessPiece(ChessGame.TeamColor.WHITE, ChessPiece.PieceType.KNIGHT));
63           addPiece(new ChessPosition(8, 2), new ChessPiece(ChessGame.TeamColor.BLACK, ChessPiece.PieceType.KNIGHT));
64           addPiece(new ChessPosition(8, 7), new ChessPiece(ChessGame.TeamColor.BLACK, ChessPiece.PieceType.KNIGHT));
65   
66           // set bishops
67           addPiece(new ChessPosition(1, 3), new ChessPiece(ChessGame.TeamColor.WHITE, ChessPiece.PieceType.BISHOP));
68           addPiece(new ChessPosition(1, 6), new ChessPiece(ChessGame.TeamColor.WHITE, ChessPiece.PieceType.BISHOP));
69           addPiece(new ChessPosition(8, 3), new ChessPiece(ChessGame.TeamColor.BLACK, ChessPiece.PieceType.BISHOP));
70           addPiece(new ChessPosition(8, 6), new ChessPiece(ChessGame.TeamColor.BLACK, ChessPiece.PieceType.BISHOP));
71   
72           // set queens
73           addPiece(new ChessPosition(8, 4), new ChessPiece(ChessGame.TeamColor.BLACK, ChessPiece.PieceType.QUEEN));
74           addPiece(new ChessPosition(1, 4), new ChessPiece(ChessGame.TeamColor.WHITE, ChessPiece.PieceType.QUEEN));
75   
76           // set kings
77           addPiece(new ChessPosition(8, 5), new ChessPiece(ChessGame.TeamColor.BLACK, ChessPiece.PieceType.KING));
78           addPiece(new ChessPosition(1, 5), new ChessPiece(ChessGame.TeamColor.WHITE, ChessPiece.PieceType.KING));
79       }

81       @Override
82       public boolean equals(Object o) {
83           if (o == null || getClass() != o.getClass()) {
84               return false;
85           }
86           ChessBoard that = (ChessBoard) o;
87           return Objects.deepEquals(board, that.board);
88       }
89   
90       @Override
91       public int hashCode() {
92           return Arrays.deepHashCode(board);
93       }

95   }
